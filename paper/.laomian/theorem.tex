\newtheorem*{Laomian@proof}{Proof}
\def\proof{\@ifnextchar[\Laomian@proof@with@name\Laomian@proof@without@name}
\def\Laomian@proof@with@name[#1]{%
  \begingroup%
  \pushQED{\qed}%
  \begin{Laomian@proof}[#1]%
}
\def\Laomian@proof@without@name{%
  \begingroup%
  \pushQED{\qed}%
  \begin{Laomian@proof}%
}
\def\endproof{%
  \popQED%
  \end{Laomian@proof}%
  \endgroup%
}
